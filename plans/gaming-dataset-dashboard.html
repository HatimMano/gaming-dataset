Base Collector Implementation

from abc import ABC, abstractmethod
import aiohttp
import asyncio
from typing import List, Dict

class BaseCollector(ABC):
    def __init__(self, config: Dict):
        self.config = config
        self.session = None
        self.rate_limiter = RateLimiter(
            calls=config.get('rate_limit', 100),
            period=60
        )
        
    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self
        
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        await self.session.close()
        
    @abstractmethod
    async def extract(self) -> List[Dict]:
        """Extract data from source"""
        pass
        
    @abstractmethod
    def parse(self, raw_data: Any) -> Dict:
        """Parse raw data into standard format"""
        pass
        
    def validate(self, data: Dict) -> bool:
        """Validate data quality"""
        required_fields = ['content', 'metadata', 'source']
        return all(field in data for field in required_fields)
        
    async def process_batch(self, items: List) -> List[Dict]:
        """Process items in parallel with rate limiting"""
        results = []
        for batch in self.chunk_list(items, 10):
            tasks = [self.process_item(item) for item in batch]
            batch_results = await asyncio.gather(*tasks)
            results.extend(batch_results)
            await self.rate_limiter.acquire()
        return results

Wikipedia Collector

class WikipediaCollector(BaseCollector):
    def __init__(self):
        super().__init__({
            'rate_limit': 200,
            'base_url': 'https://en.wikipedia.org/w/api.php'
        })
        self.categories = [
            'Category:Video_games_by_year',
            'Category:Windows_games',
            'Category:PlayStation_5_games',
            'Category:Steam_games',
            'Category:Esports_games'
        ]
        
    async def extract_category(self, category: str) -> List[str]:
        """Extract all pages from a category"""
        params = {
            'action': 'query',
            'list': 'categorymembers',
            'cmtitle': category,
            'cmlimit': 500,
            'format': 'json'
        }
        
        async with self.session.get(
            self.config['base_url'], 
            params=params
        ) as response:
            data = await response.json()
            return [
                page['pageid'] 
                for page in data['query']['categorymembers']
            ]
            
    async def extract_page_content(self, page_id: str) -> Dict:
        """Get full page content with metadata"""
        params = {
            'action': 'query',
            'prop': 'revisions|categories|info',
            'pageids': page_id,
            'rvprop': 'content',
            'format': 'json'
        }
        
        async with self.session.get(
            self.config['base_url'],
            params=params
        ) as response:
            data = await response.json()
            page = data['query']['pages'][str(page_id)]
            
            return self.parse({
                'title': page['title'],
                'content': page['revisions'][0]['*'],
                'categories': [
                    cat['title'] 
                    for cat in page.get('categories', [])
                ],
                'url': f"https://en.wikipedia.org/?curid={page_id}"
            })

Quality Scoring Formula

Validation Pipeline

class QualityScorer:
    def __init__(self):
        self.gaming_terms = self.load_gaming_dictionary()
        self.weights = {
            'length': 0.15,
            'gaming_density': 0.35,
            'structure': 0.20,
            'uniqueness': 0.20,
            'freshness': 0.10
        }
        
    def score_document(self, doc: Dict) -> float:
        scores = {
            'length': self._score_length(doc),
            'gaming_density': self._score_gaming_density(doc),
            'structure': self._score_structure(doc),
            'uniqueness': self._score_uniqueness(doc),
            'freshness': self._score_freshness(doc)
        }
        
        final_score = sum(
            scores[k] * self.weights[k] 
            for k in scores
        )
        return round(final_score, 3)
    
    def _score_gaming_density(self, doc: Dict) -> float:
        text = doc['content']['text_clean'].lower()
        words = text.split()
        gaming_words = sum(
            1 for word in words 
            if word in self.gaming_terms
        )
        return min(gaming_words / len(words) * 10, 1.0)
